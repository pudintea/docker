Network
=======
Sebelumnya kita sudah buat Network "backend-net", untuk ke minIO

> mkdir -p wordpress/web
> mkdir -p wordpress/web/html
> mkdir -p wordpress/web/nginx
> mkdir -p wordpress/phpmyadmin
> cd wordpress/web

BUAT .ENV
-------------------------
> nano .env
ADVMO_MINIO_KEY=xxx
ADVMO_MINIO_SECRET=xxx
ADVMO_MINIO_BUCKET=xxx
ADVMO_MINIO_DOMAIN=xxx
ADVMO_MINIO_ENDPOINT=xxx
ADVMO_MINIO_PATH_STYLE_ENDPOINT=false
ADVMO_MINIO_REGION=us-east-1

MYSQL_DATABASE=wordpress
MYSQL_USER=wpuser
MYSQL_PASSWORD=wppassword
MYSQL_ROOT_PASSWORD=rootpassword



> nano docker-compose.yml
Masukan :
services:
  wordpress_db:
    image: mariadb:10.11
    restart: always
    command: --character-set-server=utf8mb4 --collation-server=utf8mb4_unicode_ci
    environment:
      MYSQL_DATABASE: ${MYSQL_DATABASE}
      MYSQL_USER: ${MYSQL_USER}
      MYSQL_PASSWORD: ${MYSQL_PASSWORD}
      MYSQL_ROOT_PASSWORD: ${MYSQL_ROOT_PASSWORD}
    volumes:
      - wordpress_db_data:/var/lib/mysql
    networks:
      - backend-net

  wordpress:
    image: wordpress:php8.2-fpm
    restart: always
    volumes:
      - ./html:/var/www/html
    environment:
      WORDPRESS_DB_HOST: wordpress_db:3306
      WORDPRESS_DB_USER: ${MYSQL_USER}
      WORDPRESS_DB_PASSWORD: ${MYSQL_PASSWORD}
      WORDPRESS_DB_NAME: ${MYSQL_DATABASE}
      ADVMO_MINIO_KEY: ${ADVMO_MINIO_KEY}
      ADVMO_MINIO_SECRET: ${ADVMO_MINIO_SECRET}
      ADVMO_MINIO_BUCKET: ${ADVMO_MINIO_BUCKET}
      ADVMO_MINIO_DOMAIN: ${ADVMO_MINIO_DOMAIN}
      ADVMO_MINIO_ENDPOINT: ${ADVMO_MINIO_ENDPOINT}
      ADVMO_MINIO_PATH_STYLE_ENDPOINT: ${ADVMO_MINIO_PATH_STYLE_ENDPOINT}
      ADVMO_MINIO_REGION: ${ADVMO_MINIO_REGION}
      PHP_UPLOAD_MAX_FILESIZE: 64M
      PHP_POST_MAX_SIZE: 64M
    networks:
      - backend-net
    depends_on:
      - wordpress_db

  nginx:
    image: nginx:latest
    restart: always
    ports:
      - "8081:80"
    volumes:
      - ./html:/var/www/html
      - ./nginx/default.conf:/etc/nginx/conf.d/default.conf
    depends_on:
      - wordpress
    networks:
      - backend-net

volumes:
  wordpress_db_data:
networks:
  backend-net:
    external: true


---------------------------
> nano nginx/default.conf

server {
    listen 80;
    server_name localhost;

    root /var/www/html;
    index index.php index.html;

    location / {
        try_files $uri $uri/ /index.php?$args;
    }

    location ~ \.php$ {
        fastcgi_pass wordpress:9000;
        fastcgi_index index.php;
        include fastcgi_params;
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
    }
}

RUN
===
Jalankan compose dengan env file
Gunakan:
> docker compose --env-file .env up -d

Hapus volume bila rusak.
Jalankan:
> docker compose down -v


PHPMYADMIN
---------------------------------
> cd ../phpmyadmin
> nano docker-compose.yml
Masukan :
services:
  wp_phpmyadmin:
    image: phpmyadmin/phpmyadmin
    container_name: wp_phpmyadmin
    restart: always
    environment:
      PMA_HOST: wordpress_db
    ports:
      - "8082:80"
    networks:
      - backend-net
networks:
  backend-net:
    external: true
